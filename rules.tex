%%%These are handy short hands for each of the inference rules.

%And elimination
\newcommand{\andE}[1]
{
$\land \mathrm{E}(#1)$
}

%And introduction
\newcommand{\andI}[2]
{
$\land \mathrm{I}(#1,#2)$
}

%Assume
\newcommand{\ass} {\sout{Assume}}

%Given
\newcommand{\giv} {Given}

%See lemma command \slem{key}{line applied to}
\newcommand{\alem}[2]
{
Apply Lemma \ref{#1} to #2
}

%not not introduction
\newcommand{\nnI}[1]
{
$ \lnot\lnot \mathrm{I}(#1) $
}

%or elimination
\newcommand{\orE}[2]
{
$ \lor \mathrm{E}(#1,#2) $
}

%or introduction
\newcommand{\orI}[1]
{
$ \lor \mathrm{I}(#1) $
}

%arrow introduction
\newcommand{\arrowI}[2]
{
$\to \mathrm{I}(#1,#2) $
}

%arrow eliminate
\newcommand{\arrowE}[2]
{
$\to \mathrm{E}(#1,#2) $
}

%double arrow eliminate
\newcommand{\iffE}[1]
{
$\leftrightarrow \mathrm{E}(#1)$
}


% Reductio Ad Absurdum
\newcommand{\raa}[3]
{
RAA(#1,#2,#3)
}

% There Exists Introduction
\newcommand{\existsI}[1]
{
$ \exists \mathrm{I}(#1) $
}

% There Exists Elimination
\newcommand{\existsE}[3]
{
$ \exists \mathrm{E}(#1,#2,#3) $
}

% For All Introduction
\newcommand{\forallI}[1]
{
$ \forall \mathrm{I}(#1) $
}

% For All Elimination
\newcommand{\forallE}[1]
{
$ \forall \mathrm{E}(#1) $
}
